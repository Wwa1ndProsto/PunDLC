    @Override
    public void onEvent(Event event) {
        ClientPlayerEntity player = mc.player;
        if (player == null || player.isDead()) {
            reset();
            return;
        }

        if (event instanceof EventKeyBoard e && correction.get() && correctionType.is("Free")) {
            MoveUtil.fixMovement(e, Manager.FUNCTION_MANAGER.autoPotion.isActivePotion ?
                    Manager.ROTATION.getPitch() : Manager.ROTATION.getYaw());
        }

        if (event instanceof EventSprint sprint && sprintReset.is("Legit")) {
            // сброс спринта в attackTarget
        }

        if (event instanceof EventUpdate) {
            updateTargetCache();

            if (mode.is("MultiAura")) {
                updateMultiTargets();
            } else {
                if (target == null || !isValidTarget(target)) {
                    target = findBestTarget();
                }
            }

            if ((mode.is("MultiAura") && multiTargets.isEmpty()) || (!mode.is("MultiAura") && target == null)) {
                Manager.ROTATION.set(player.getYaw(), player.getPitch());
                cpsLimit = System.currentTimeMillis();
                return;
            }

            handleAttackAndRotation();
        }

        if (event instanceof EventMotion motion) {
            motion.setYaw(Manager.ROTATION.getYaw());
            motion.setPitch(Manager.ROTATION.getPitch());
        }

        // Обработка кнопок записи (с защитой от повторных вызовов)
        if (startRecording.get()) {
            startRecording.setValue(false);
            startRecordingAction();
        }
        if (stopRecording.get()) {
            stopRecording.setValue(false);
            stopRecordingAction();
        }
        if (deleteRecording.get()) {
            deleteRecording.setValue(false);
            deleteRecordingAction();
        }
    }

    @Override
    protected void onEnable() {
        super.onEnable();
        reset();
        neuroEngine.reset();
        updateRecordingList();

        if (mode.is("Recorded") && !recorder.isPlaying()) {
            if (!selectedRecording.get().equals("Выберите ротацию") && !selectedRecording.get().isEmpty()) {
                recorder.startPlayback(selectedRecording.get());
            } else if (mc.player != null) {
                mc.player.sendMessage(Text.literal("§c[AttackAura] Выберите ротацию в настройках для воспроизведения"), false);
            }
        }
    }

    @Override
    protected void onDisable() {
        if (recorder.isRecording()) recorder.stopRecording();
        if (recorder.isPlaying()) recorder.stopPlayback();
        if (target != null && isValidTarget(target)) {
            String modeName = mode.get();
            if (modeName.equals("FunTime") || modeName.equals("HollyWorld") || modeName.equals("ReallyWorld")) {
                Manager.ROTATION.smoothReturn(350);
            } else {
                Manager.ROTATION.set(mc.player.getYaw(), mc.player.getPitch());
            }
        }
        reset();
        neuroEngine.reset();
        super.onDisable();
    }

        target = null;

    // -------------------- Настройки записи/воспроизведения --------------------
    private void updateRecordingList() {
        List<String> list = recorder.listRecordings();
        if (list.isEmpty()) {
            selectedRecording.setValues(new String[]{"Выберите ротацию"});
            selectedRecording.set("Выберите ротацию");
        } else {
            String[] arr = list.toArray(new String[0]);
            selectedRecording.setValues(arr);
            if (!selectedRecording.get().equals("Выберите ротацию") && !list.contains(selectedRecording.get())) {
                selectedRecording.set(arr[0]);
            } else if (selectedRecording.get().equals("Выберите ротацию")) {
                selectedRecording.set(arr[0]);
            }
        }
    }

    private void startRecordingAction() {
        if (recorder.isRecording()) {
            // уже запись, не начинаем новую
            return;
        }
        String name = recordingName.get();
        if (name == null || name.isEmpty()) {
            mc.player.sendMessage(Text.literal("§c[AttackAura] Введите имя ротации"), false);
            return;
        }
        recorder.startRecording(name);
        mc.player.sendMessage(Text.literal("§a[AttackAura] Начата запись: " + name), false);
    }

    private void stopRecordingAction() {
        if (!recorder.isRecording()) {
            return;
        }
        String lastRecording = recorder.getLastRecordingName();
        recorder.stopRecording();
        if (lastRecording != null) {
            mc.player.sendMessage(Text.literal("§a[AttackAura] Запись сохранена: " + lastRecording), false);
            updateRecordingList();
        } else {
            mc.player.sendMessage(Text.literal("§c[AttackAura] Ошибка: имя ротации не определено"), false);
        }
    }

    private void deleteRecordingAction() {
        String name = selectedRecording.get();
        if (name.equals("Выберите ротацию") || name.isEmpty()) {
            mc.player.sendMessage(Text.literal("§c[AttackAura] Сначала выберите ротацию"), false);
            return;
        }
        recorder.deleteRecording(name);
        updateRecordingList();
        mc.player.sendMessage(Text.literal("§a[AttackAura] Ротация удалена: " + name), false);
    }

    // -------------------- Поиск и валидация целей --------------------
    private void updateTargetCache() {
        if (cacheTicker++ % CACHE_REFRESH_INTERVAL != 0) return;

        List<LivingEntity> newList = new ArrayList<>();
        for (Entity e : Manager.SYNC_MANAGER.getEntities()) {
            if (e instanceof LivingEntity le && isValidTarget(le)) {
                newList.add(le);
            }
        }
        cachedEntities = newList;
    }

    private boolean isValidTarget(LivingEntity entity) {
        if (entity == null || entity.isDead() || !entity.isAlive() || entity == mc.player) return false;

        double dist = AuraUtil.getDistance(entity);
        double attackRange = distance.get().doubleValue();
        double detectRange = mc.player.isGliding() ? elytraDistance.get().doubleValue() : rotateDistance.get().doubleValue();

        if (dist > attackRange && (detectRange <= 0 || dist > detectRange)) return false;
        if (Manager.FUNCTION_MANAGER.antiBot.check(entity)) return false;

        if (entity instanceof PlayerEntity) {
            if (!targets.get("Игроки")) return false;
            if (!targets.get("Друзья") && Manager.FRIEND_MANAGER.isFriend(entity.getName().getString())) return false;
        } else if (entity instanceof VillagerEntity && !targets.get("Жители")) return false;
        else if (entity instanceof MobEntity || entity instanceof AnimalEntity) {
            if (!targets.get("Мобы")) return false;
        } else if (entity instanceof Monster && !targets.get("Монстры")) return false;

        if (entity instanceof ArmorStandEntity) return false;
        if (Manager.FUNCTION_MANAGER.littleSnickers.state && (entity instanceof GhostWolfEntity)) return false;

        if (entity instanceof PlayerEntity p && (p.isInvisible() || p.isCreative())) return false;

        return true;
    }

    private LivingEntity findBestTarget() {
        if (cachedEntities.isEmpty()) return null;

        Comparator<LivingEntity> comparator;
        switch (sort.get()) {
            case "По здоровью":
                comparator = Comparator.comparing(LivingEntity::getHealth);
                break;
            case "По дистанции":
                comparator = Comparator.comparingDouble(mc.player::distanceTo);
                break;
            case "По броне":
                comparator = Comparator.comparingDouble(AuraUtil::getArmor);
                break;
            case "По углу":
                comparator = Comparator.comparingDouble(this::getAngleToEntity);
                break;
            case "По угрозе":
                comparator = Comparator.comparingDouble(this::getThreatLevel).reversed();
                break;
            default:
                comparator = Comparator.comparingDouble(mc.player::distanceTo);
        }

        return cachedEntities.stream().min(comparator).orElse(null);
    }

    private void updateMultiTargets() {
        multiTargets.clear();
        int max = maxTargets.get().intValue();
        if (max <= 0) return;

        List<LivingEntity> sorted = new ArrayList<>(cachedEntities);
        sorted.sort(Comparator.comparingDouble(mc.player::distanceTo));
        for (int i = 0; i < Math.min(max, sorted.size()); i++) {
            multiTargets.add(sorted.get(i));
        }
    }

    // -------------------- Обработка атаки и ротации --------------------
    private void handleAttackAndRotation() {
        boolean canAttackNow = canAttack();
        boolean noPotion = !Manager.FUNCTION_MANAGER.autoPotion.isActivePotion;

        // Режим Recorded (воспроизведение записи)
        if (mode.is("Recorded")) {
            if (recorder.isPlaying()) {
                RotationFrame frame = recorder.getCurrentPlaybackFrame();
                if (frame != null) {
                    Manager.ROTATION.setSmooth(frame.yaw, frame.pitch, 0.9f, 120f, 20f, true);
                    if (frame.attack && canAttackNow && checkRaycast(target) && noPotion && target != null) {
                        attackTarget(target);
                    }
                } else {
                    recorder.stopPlayback();
                }
            } else if (autoUpdateList.get() && !selectedRecording.get().equals("Выберите ротацию")) {
                recorder.startPlayback(selectedRecording.get());
            }
        }

        // MultiAura
        if (mode.is("MultiAura")) {
            for (LivingEntity e : multiTargets) {
                if (shouldAttack(e) && canAttackNow && checkRaycast(e) && noPotion) attackTarget(e);
            }
            if (!multiTargets.isEmpty()) setRotation(multiTargets.get(0), true);
            return;
        }

        if (target == null) return;

        // Элитры
        if (handleElytraRotation(target)) {
            if (shouldAttack(target) && canAttackNow && checkRaycast(target) && noPotion) attackTarget(target);
            return;
        }

        String modeName = mode.get();
        switch (modeName) {
            case "NeuroUltra":
                handleNeuroUltra(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "KoopinAc":
            case "1.8.8":
                koopinVector(target, true);
                if (shouldAttack(target) && canAttackNow && checkRaycast(target) && noPotion) attackTarget(target);
                break;
            case "SpookyTime":
                handleSpookyTime(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "LonyGrief":
                handleLonyGrief(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "FunTime":
                handleFunTime(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "HollyWorld":
                handleHollyWorld(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "Snap":
                handleSnap(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "Switch":
                handleSwitchMode(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "Legit":
                handleLegit(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "Silent":
                handleSilent(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            case "Velocity":
                handleVelocity(target, canAttackNow && checkRaycast(target) && noPotion);
                break;
            default:
                if (shouldAttack(target) && canAttackNow && checkRaycast(target) && noPotion) attackTarget(target);
                setRotation(target, true);
                break;
        }

        // ЗАПИСЬ КАДРОВ (работает в любом режиме, если запись активна)
        if (recorder.isRecording()) {
            float yaw = Manager.ROTATION.getYaw();
            float pitch = Manager.ROTATION.getPitch();
            boolean attacking = canAttackNow && target != null && checkRaycast(target) && noPotion;
            recorder.recordFrame(yaw, pitch, attacking);
        }
    }

    // -------------------- Нейро-режим ULTRA --------------------
    private void handleNeuroUltra(LivingEntity t, boolean attack) {
        neuroEngine.update(t);
        Vec3d predictedPoint = neuroEngine.getPredictedPoint(t, attack);
        Vec2f rot = getRotationTo(predictedPoint);
        float smooth = 0.85f;
        if (neuroJitterAmount.get().floatValue() > 0) {
            rot = new Vec2f(
                    rot.x + (random.nextFloat() - 0.5f) * neuroJitterAmount.get().floatValue() * 5,
                    rot.y + (random.nextFloat() - 0.5f) * neuroJitterAmount.get().floatValue() * 5
            );
        }
        Manager.ROTATION.setSmooth(rot.y, rot.x, smooth, 120f, 20f, true);
        if (attack) {
            attackTarget(t);
            neuroEngine.onAttackSuccess(t);
        } else {
            neuroEngine.onAttackMiss(t);
        }
    }

    // -------------------- Режимы (SpookyTime, LonyGrief, FunTime, HollyWorld, Snap, Switch, Legit, Silent, Velocity, koopinVector) --------------------
    private void handleSpookyTime(LivingEntity t, boolean attack) {
        if (mc.player == null) return;

        if (target == null) {
            randomYawOffset = 0;
            randomPitchOffset = 0;
            float centerYaw = mc.player.getYaw();
            float centerPitch = mc.player.getPitch();

            prevBodyYaw = bodyYaw;
            prevBodyPitch = bodyPitch;
            float yawDiff = MathHelper.wrapDegrees(centerYaw - bodyYaw);
            float pitchDiff = centerPitch - bodyPitch;
            bodyYaw += MathHelper.clamp(yawDiff, -45, 45);
            bodyPitch = MathHelper.clamp(bodyPitch + pitchDiff, -90f, 90f);

            prevHeadYaw = headYaw;
            prevHeadPitch = headPitch;
            float yawDiffH = MathHelper.wrapDegrees(centerYaw - headYaw);
            float pitchDiffH = centerPitch - headPitch;
            headYaw += MathHelper.clamp(yawDiffH, -50, 50);
            headPitch = MathHelper.clamp(headPitch + pitchDiffH, -90f, 90f);
            return;
        }

        randomUpdateTicks++;
        if (randomUpdateTicks >= updateInterval) {
            randomUpdateTicks = 0;
            randomYawOffset = (random.nextFloat() * 2 - 1) * maxYawShake;
            randomPitchOffset = (random.nextFloat() * 2 - 1) * maxPitchShake;
        }

        Vec3d targetPos = getRandomBodyPoint(t);
        Vec2f rot = getRotationTo(targetPos);

        prevBodyYaw = bodyYaw;
        prevBodyPitch = bodyPitch;
        float yawDiff = MathHelper.wrapDegrees(rot.y - bodyYaw);
        float pitchDiff = rot.x - bodyPitch;
        bodyYaw += MathHelper.clamp(yawDiff, -45, 45);
        bodyPitch = MathHelper.clamp(bodyPitch + pitchDiff, -45, 45);

        prevHeadYaw = headYaw;
        prevHeadPitch = headPitch;
        float yawDiffH = MathHelper.wrapDegrees(rot.y - headYaw);
        float pitchDiffH = rot.x - headPitch;
        headYaw += MathHelper.clamp(yawDiffH, -50, 50);
        headPitch = MathHelper.clamp(headPitch + pitchDiffH, -50, 50);

        Vec2f finalRot = new Vec2f(rot.x + randomPitchOffset, rot.y + randomYawOffset);
        Manager.ROTATION.setSmooth(finalRot.y, finalRot.x, 0.8f, 60, 90, true);

        if (attack) attackTarget(t);
    }

    private void handleLonyGrief(LivingEntity t, boolean attack) {
        Vec3d tp = predictPos(t);
        double yawToTarget = Math.toDegrees(Math.atan2(tp.z - mc.player.getZ(), tp.x - mc.player.getX())) - 90.0;
        double yawDiff = Math.abs(MathHelper.wrapDegrees((float) yawToTarget - Manager.ROTATION.getYaw()));
        if (yawDiff <= 180 && attack) attackTarget(t);
        Manager.ROTATION.set(mc.player.getYaw(), mc.player.getPitch());
    }

    private void handleFunTime(LivingEntity t, boolean attack) {
        if (attack) {
            attackTarget(t);
            lastAttackMs = System.currentTimeMillis();
        }

        if (System.currentTimeMillis() - lastAttackMs < 450) {
            Vec3d eye = mc.player.getEyePos();
            Vec3d base = t.getPos();
            float[] points = new float[]{0.82f, 0.67f, 0.43f, 0.27f};
            float mul = points[(int) (System.currentTimeMillis() / 180 % points.length)];
            Vec3d targetPos = new Vec3d(base.x, base.y + t.getHeight() * mul, base.z);

            double halfWidth = t.getWidth() / 2.0;
            double sideOffset = swingSideRight ? halfWidth * 1.2f : -halfWidth;
            double yawToEntity = Math.atan2(t.getZ() - mc.player.getZ(), t.getX() - mc.player.getX());
            double offsetX = Math.cos(yawToEntity + Math.PI / 2) * sideOffset;
            double offsetZ = Math.sin(yawToEntity + Math.PI / 2) * sideOffset;
            targetPos = targetPos.add(offsetX, 0, offsetZ);

            Vec2f rot = getRotationTo(targetPos);
            long now = System.currentTimeMillis();

            if (now - lastSwitch > 200 + random.nextInt(250)) {
                lastSwitch = now;
                swingSideRight = !swingSideRight;
                float distanceFactor = (float) MathHelper.clamp(mc.player.distanceTo(t) / 6.0f, 0.4f, 1.0f);
                float maxDeviation = 4.0f * distanceFactor;
                jitterYawTarget = (swingSideRight ? maxDeviation : -maxDeviation) + (float) (random.nextGaussian() * 0.6f);
            }

            float diff = jitterYawTarget - jitterYaw;
            jitterYawSpeed += diff * 0.05f;
            jitterYawSpeed *= 0.88f;
            jitterYaw += jitterYawSpeed;
            jitterYaw *= 0.985f;

            if (now - lastBreathChange > 2000 + random.nextInt(1500)) {
                lastBreathChange = now;
                swaySpeed = 0.035f + random.nextFloat() * 0.02f;
                swayAmplitude = 2.0f + random.nextFloat() * 1.2f;
            }

            swayPhase += swaySpeed;
            float sway = (float) Math.sin(swayPhase) * swayAmplitude;
            float totalYawOffset = (float) MathHelper.clamp(jitterYaw + sway, -halfWidth * 8.5f, halfWidth * 8.5f);
            microJitter += (random.nextFloat() - 0.5f) * 0.25f;
            microJitter *= 0.85f;

            float finalYaw = rot.y + totalYawOffset + microJitter;
            float finalPitch = rot.x + (float) Math.sin(swayPhase * 0.8f) * 0.5f;
            Manager.ROTATION.setSmooth(finalYaw, finalPitch, 1.1f, 180f, 15f, true);
        } else {
            long currentTime = System.currentTimeMillis();
            if (shakeStartTime == 0L) shakeStartTime = currentTime;
            float elapsedSec = (currentTime - shakeStartTime) / 1000f;
            double angle = 2 * Math.PI * 2.4 * elapsedSec;
            float yawOffset = (float) Math.sin(angle) * 24f;
            double angle2 = 2 * Math.PI * 0.08f * elapsedSec;
            double[] options = {5.0, 5.5, 5.8, 6.0};
            double randAmplitude = options[(int) (Math.random() * options.length)];
            float yawOffset2 = (float) (Math.sin(angle2) * randAmplitude);
            float finalYaw = mc.player.getYaw() + yawOffset + yawOffset2;
            float finalPitch = 0.0f + yawOffset2;
            Manager.ROTATION.setSmooth(finalYaw, finalPitch, 1.0f, 20f, 10f, true);
        }
    }

    private Vec2f getRotationTo(Vec3d point) {
        Vec3d eye = mc.player.getEyePos();
        double dx = point.x - eye.x;
        double dy = point.y - eye.y;
        double dz = point.z - eye.z;
        double hDist = Math.sqrt(dx * dx + dz * dz);
        float yaw = (float) (Math.toDegrees(Math.atan2(dz, dx)) - 90.0);
        float pitch = (float) -Math.toDegrees(Math.atan2(dy, hDist));
        return new Vec2f(pitch, yaw);
    }

    private Vec3d predictPos(LivingEntity e) {
        Vec3d currentPos = e.getPos();
        Vec3d velocity = e.getVelocity();

        var ts = Manager.FUNCTION_MANAGER.targetStrafe;
        if (ts.state && ts.predictCheck.get()) {
            float pr = ts.predict.get().floatValue();
            if (pr > 0) {
                return currentPos.add(velocity.multiply(pr));
            }
        }

        int ticks = predictTicks.get().intValue();
        if (ticks <= 0) return currentPos;

        if (lastTargetPos != null) {
            velocity = currentPos.subtract(lastTargetPos);
            lastTargetVelocity = velocity;
        }
        lastTargetPos = currentPos;

        Vec3d predicted = currentPos.add(velocity.multiply(ticks));
        double maxPredict = e.getBoundingBox().getAverageSideLength() * ticks;
        if (predicted.distanceTo(currentPos) > maxPredict) {
            predicted = currentPos.add(velocity.normalize().multiply(maxPredict));
        }
        return predicted;
    }

    private Vec3d getRandomBodyPoint(LivingEntity entity) {
        double x = entity.getBoundingBox().getCenter().x;
        double y = entity.getY();
        double z = entity.getBoundingBox().getCenter().z;

        int randPoint = random.nextInt(4);
        switch (randPoint) {
            case 0 -> y += entity.getHeight() * 0.9;
            case 1 -> y += entity.getHeight() * 0.75;
            case 2 -> y += entity.getHeight() * 0.5;
            case 3 -> y += entity.getHeight() * 0.25;
        }

        x += (random.nextDouble() * 0.4 - 0.2);
        z += (random.nextDouble() * 0.4 - 0.2);

        return new Vec3d(x, y, z);
    }

    private void koopinVector(LivingEntity entity, boolean attackContext) {
        Vec3d head = entity.getEyePos().add(0, entity.getHeight(), 0);
        Vec3d chest = entity.getEyePos().add(0, entity.getStandingEyeHeight() / 2.0f, 0);
        Vec3d legs = entity.getEyePos().add(0, 0.05, 0);
        Vec3d[] points = new Vec3d[]{head, chest, legs};

        float bestPitchDelta = Float.MAX_VALUE;
        Vec3d best = chest;
        float currPitch = Manager.ROTATION.getPitch();
        float currYaw = Manager.ROTATION.getYaw();

        for (Vec3d p : points) {
            Vec2f rot = getRotationTo(p);
            float pitchDelta = Math.abs(rot.x - currPitch);
            if (pitchDelta < bestPitchDelta) {
                bestPitchDelta = pitchDelta;
                best = p;
            }
        }

        Vec2f rot = getRotationTo(best);
        float yawDelta = MathHelper.wrapDegrees(rot.y - currYaw);
        float pitchDelta = rot.x - currPitch;

        float addYaw = Math.min(Math.max(Math.abs(yawDelta), 1), 80);
        if (Math.abs(addYaw) <= 3.0f) addYaw = 3.1f;
        float addPitch = Math.max(attackContext ? Math.abs(pitchDelta) : 1.0f, 2.0f);

        float ny = currYaw + (yawDelta > 0 ? addYaw : -addYaw);
        float np = MathHelper.clamp(currPitch + (pitchDelta > 0 ? addPitch : -addPitch), -90.0f, 90.0f);
        Manager.ROTATION.set(ny, np);
    }

    private void attackTarget(LivingEntity e) {
        if (e == null) return;

        boolean sprintStop = false;
        boolean canStartSprint = mc.player.input.movementForward > 0
                && !mc.player.hasStatusEffect(StatusEffects.BLINDNESS)
                && !mc.player.isGliding()
                && !mc.player.isUsingItem()
                && !mc.player.horizontalCollision
                && mc.player.getHungerManager().getFoodLevel() > 6
                && !mc.player.isSneaking();

        if (setting.get("Отжим щита") && mc.player.isBlocking()) {
            mc.interactionManager.stopUsingItem(mc.player);
        }

        if (sprintReset.is("Rage")) {
            if (((ClientPlayerEntityAccessor) mc.player).getLastSprinting()) {
                mc.player.networkHandler.sendPacket(new ClientCommandC2SPacket(mc.player, ClientCommandC2SPacket.Mode.STOP_SPRINTING));
                mc.player.setSprinting(false);
                sprintStop = true;
            }
        }

        cpsLimit = System.currentTimeMillis() + attackDelay.get().longValue();
        lastAttackMs = System.currentTimeMillis();

        mc.interactionManager.attackEntity(mc.player, e);
        mc.player.swingHand(MAIN_HAND);

        if (setting.get("Ломать щит") && e instanceof PlayerEntity) {
            shieldBreaker((PlayerEntity) e);
        }

        ElytraTarget elytraTarget = Manager.FUNCTION_MANAGER.elytraTarget;
        if (elytraTarget.mode.is("Продвинутый")) {
            elytraTarget.trueFireWork = true;
            if (elytraTarget.prefer.get()) elytraTarget.nextPhase(e);
        }

        if (sprintReset.is("Rage") && sprintStop && canStartSprint) {
            if (((ClientPlayerEntityAccessor) mc.player).getLastSprinting()) {
                mc.player.networkHandler.sendPacket(new ClientCommandC2SPacket(mc.player, ClientCommandC2SPacket.Mode.START_SPRINTING));
                mc.player.setSprinting(true);
            }
        }
    }

    private boolean shieldBreaker(PlayerEntity targetPlayer) {
        int axeSlot = InventoryUtil.getAxe().slot();
        if (axeSlot == -1) return false;
        if (!targetPlayer.isUsingItem()) return false;
        if (targetPlayer.getOffHandStack().getItem() != Items.SHIELD
                && targetPlayer.getMainHandStack().getItem() != Items.SHIELD) return false;

        if (axeSlot >= 9) {
            mc.interactionManager.clickSlot(mc.player.currentScreenHandler.syncId, axeSlot,
                    mc.player.getInventory().selectedSlot, SlotActionType.SWAP, mc.player);
            mc.getNetworkHandler().sendPacket(new CloseHandledScreenC2SPacket(mc.player.currentScreenHandler.syncId));
            mc.interactionManager.attackEntity(mc.player, targetPlayer);
            mc.player.swingHand(Hand.MAIN_HAND);
            mc.interactionManager.clickSlot(mc.player.currentScreenHandler.syncId, axeSlot,
                    mc.player.getInventory().selectedSlot, SlotActionType.SWAP, mc.player);
            mc.getNetworkHandler().sendPacket(new CloseHandledScreenC2SPacket(mc.player.currentScreenHandler.syncId));
        } else {
            mc.getNetworkHandler().sendPacket(new UpdateSelectedSlotC2SPacket(axeSlot));
            mc.interactionManager.attackEntity(mc.player, targetPlayer);
            mc.player.swingHand(Hand.MAIN_HAND);
            mc.getNetworkHandler().sendPacket(new UpdateSelectedSlotC2SPacket(mc.player.getInventory().selectedSlot));
        }
        return true;
    }

    private boolean shouldAttack(LivingEntity e) {
        if (e == null) return false;
        if (cpsLimit > System.currentTimeMillis()) return false;
        if (AuraUtil.getDistance(e) > distance.get().doubleValue()) return false;
        return canAttack();
    }

    private boolean canAttack() {
        if (noAttackIfEat.get() && mc.player.isUsingItem() && !mc.player.getActiveItem().isOf(Items.SHIELD)) return false;
        if (System.currentTimeMillis() < cpsLimit) return false;

        if (!(mc.player.getMainHandStack().isOf(Items.MACE))
                && mc.player.getAttackCooldownProgress(mc.getRenderTickCounter().getTickDelta(true)) < 0.9F) return false;

        boolean restrict = mc.player.hasStatusEffect(StatusEffects.BLINDNESS)
                || mc.player.hasStatusEffect(StatusEffects.LEVITATION)
                || mc.player.hasStatusEffect(StatusEffects.SLOW_FALLING)
                || mc.player.isInLava()
                || mc.player.inPowderSnow
                || mc.player.isClimbing()
                || mc.player.hasVehicle()
                || mc.player.getAbilities().flying
                || (mc.player.isInFluid() && !mc.options.jumpKey.isPressed())
                || MoveUtil.isInWeb();

        boolean needSpace = onlySpaceCritical.get()
                && mc.player.isOnGround()
                && !mc.options.jumpKey.isPressed();

        if (setting.get("Только критами") && !restrict) {
            return needSpace || (!mc.player.isOnGround() && mc.player.fallDistance > 0.0f);
        }
        return true;
    }

    private double getAngleToEntity(LivingEntity entity) {
        Vec2f rot = getRotationTo(entity.getBoundingBox().getCenter());
        float yawDiff = MathHelper.wrapDegrees(rot.y - Manager.ROTATION.getYaw());
        float pitchDiff = rot.x - Manager.ROTATION.getPitch();
        return Math.sqrt(yawDiff * yawDiff + pitchDiff * pitchDiff);
    }

    private double getThreatLevel(LivingEntity entity) {
        double threat = 0.0;
        if (entity instanceof PlayerEntity p) {
            threat += p.getHealth() < 10 ? 5 : 0;
            threat += (p.getArmor() / 20.0) * 10;
            threat += p.getMainHandStack().getMaxDamage() / 100.0;
            if (p.isBlocking()) threat += 3;
        } else if (entity instanceof Monster) {
            threat += 2;
        }
        return threat;
    }

    // -------------------- НЕЙРО-ДВИЖОК ULTRA (полная система предсказания) --------------------
    private class NeuroEngineUltra {
        // История позиций с временными метками
        private final Deque<HistoryEntry> history = new ArrayDeque<>();
        private Vec3d lastPos = null;
        private long lastTime = 0;

        // Модели предсказания
        private final List<PredictionModel> models = new ArrayList<>();
        private double[] modelWeights;

        // Профиль противника
        private PlayerProfile profile = null;

        // Смещение (bias)
        private Vec3d bias = Vec3d.ZERO;

        // Статистика
        private int hitCount = 0;
        private int missCount = 0;

        public NeuroEngineUltra() {
            models.add(new ConstantVelocityModel());
            models.add(new ConstantAccelerationModel());
            models.add(new SmoothedDirectionModel());
            models.add(new ClusterBasedModel());
            models.add(new RecurrentModel());
            modelWeights = new double[models.size()];
            Arrays.fill(modelWeights, 1.0 / models.size());
        }

        public void update(LivingEntity target) {
            long now = System.currentTimeMillis();
            Vec3d pos = target.getPos();
            if (lastPos != null && lastTime > 0) {
                double dt = (now - lastTime) / 1000.0;
                if (dt > 0.001) {
                    history.add(new HistoryEntry(pos, now, dt));
                    while (history.size() > neuroMemory.get().intValue()) {
                        history.removeFirst();
                    }
                }
            }
            lastPos = pos;
            lastTime = now;

            if (neuroSaveProfile.get() && target instanceof PlayerEntity) {
                updateProfile(target);
            }
        }

        public Vec3d getPredictedPoint(LivingEntity target, boolean attackNow) {
            if (history.isEmpty()) return target.getPos();

            double px = 0, py = 0, pz = 0;
            double totalWeight = 0;
            for (int i = 0; i < models.size(); i++) {
                Vec3d pred = models.get(i).predict(history, target);
                px += pred.x * modelWeights[i];
                py += pred.y * modelWeights[i];
                pz += pred.z * modelWeights[i];
                totalWeight += modelWeights[i];
            }
            if (totalWeight > 0) {
                px /= totalWeight;
                py /= totalWeight;
                pz /= totalWeight;
            }
            Vec3d finalPred = new Vec3d(px, py, pz);

            // Применяем смещение и профиль
            finalPred = finalPred.add(bias);
            if (profile != null) finalPred = profile.correctPrediction(finalPred);

            Box box = target.getBoundingBox();
            double x = MathHelper.clamp(finalPred.x, box.minX, box.maxX);
            double y = MathHelper.clamp(finalPred.y, box.minY + 0.1, box.maxY - 0.1);
            double z = MathHelper.clamp(finalPred.z, box.minZ, box.maxZ);
            if (!target.isOnGround()) y += 0.2;
            return new Vec3d(x, y, z);
        }

        public void onAttackSuccess(LivingEntity target) {
            hitCount++;
            updateWeights(true);
            bias = bias.multiply(0.95);
            if (neuroSaveProfile.get() && target instanceof PlayerEntity) saveProfile(target);
        }

        public void onAttackMiss(LivingEntity target) {
            missCount++;
            updateWeights(false);
            Vec3d vel = target.getVelocity();
            if (vel.lengthSquared() > 0.01) {
                double adapt = neuroLearningRate.get().floatValue();
                bias = bias.add(vel.multiply(adapt * 0.2));
                double maxBias = 1.0;
                if (bias.length() > maxBias) bias = bias.normalize().multiply(maxBias);
            }
            if (neuroSaveProfile.get() && target instanceof PlayerEntity) saveProfile(target);
        }

        private void updateWeights(boolean success) {
            if (history.size() < 2) return;
            HistoryEntry last = history.getLast();
            Vec3d actual = last.pos;
            double[] errors = new double[models.size()];
            double minError = Double.MAX_VALUE;
            for (int i = 0; i < models.size(); i++) {
                Vec3d pred = models.get(i).predict(history, null);
                double err = pred.distanceTo(actual);
                errors[i] = err;
                if (err < minError) minError = err;
            }
            double lr = neuroLearningRate.get().floatValue();
            for (int i = 0; i < models.size(); i++) {
                if (success && errors[i] == minError) {
                    modelWeights[i] += lr;
                } else if (!success) {
                    modelWeights[i] -= lr * 0.5;
                }
                if (modelWeights[i] < 0) modelWeights[i] = 0;
            }
            double sum = Arrays.stream(modelWeights).sum();
            if (sum > 0) {
                for (int i = 0; i < modelWeights.length; i++) modelWeights[i] /= sum;
            } else {
                Arrays.fill(modelWeights, 1.0 / models.size());
            }
        }

        private void updateProfile(LivingEntity target) {
            String name = target.getName().getString();
            profile = loadProfile(name);
            if (profile == null) profile = new PlayerProfile(name);
            profile.update(target, history);
        }

        private void saveProfile(LivingEntity target) {
            if (profile != null) profile.save();
        }

        private PlayerProfile loadProfile(String name) {
            Path path = Paths.get(System.getProperty("user.dir"), "run", "profiles", name + ".json");
            if (!Files.exists(path)) return null;
            try (Reader reader = new FileReader(path.toFile())) {
                Gson gson = new Gson();
                return gson.fromJson(reader, PlayerProfile.class);
            } catch (IOException e) {
                return null;
            }
        }

        public void reset() {
            history.clear();
            lastPos = null;
            bias = Vec3d.ZERO;
            hitCount = missCount = 0;
            Arrays.fill(modelWeights, 1.0 / models.size());
            profile = null;
        }
    }

    // -------------------- Вспомогательные классы для нейро-движка --------------------
    private static class HistoryEntry {
        final Vec3d pos;
        final long time;
        final double dt;
        HistoryEntry(Vec3d pos, long time, double dt) {
            this.pos = pos;
            this.time = time;
            this.dt = dt;
        }
    }

    private interface PredictionModel {
        Vec3d predict(Deque<HistoryEntry> history, LivingEntity target);
    }

    private static class ConstantVelocityModel implements PredictionModel {
        @Override
        public Vec3d predict(Deque<HistoryEntry> history, LivingEntity target) {
            if (history.size() < 2) return target != null ? target.getPos() : Vec3d.ZERO;
            HistoryEntry prev = history.getFirst();
            HistoryEntry last = history.getLast();
            double dt = (last.time - prev.time) / 1000.0;
            if (dt <= 0) return last.pos;
            Vec3d vel = last.pos.subtract(prev.pos).multiply(1.0 / dt);
            return last.pos.add(vel.multiply(0.05));
        }
    }

    private static class ConstantAccelerationModel implements PredictionModel {
        @Override
        public Vec3d predict(Deque<HistoryEntry> history, LivingEntity target) {
            if (history.size() < 3) return target != null ? target.getPos() : Vec3d.ZERO;
            List<HistoryEntry> list = new ArrayList<>(history);
            int n = list.size();
            HistoryEntry p0 = list.get(n-3);
            HistoryEntry p1 = list.get(n-2);
            HistoryEntry p2 = list.get(n-1);
            double dt1 = p1.dt;
            double dt2 = p2.dt;
            if (dt1 <= 0 || dt2 <= 0) return p2.pos;
            Vec3d v1 = p1.pos.subtract(p0.pos).multiply(1.0 / dt1);
            Vec3d v2 = p2.pos.subtract(p1.pos).multiply(1.0 / dt2);
            Vec3d a = v2.subtract(v1).multiply(1.0 / ((p2.time - p0.time)/1000.0));
            return p2.pos.add(v2.multiply(0.05)).add(a.multiply(0.05*0.05*0.5));
        }
    }

    private static class SmoothedDirectionModel implements PredictionModel {
        @Override
        public Vec3d predict(Deque<HistoryEntry> history, LivingEntity target) {
            if (history.size() < 2) return target != null ? target.getPos() : Vec3d.ZERO;
            HistoryEntry last = history.getLast();
            Vec3d vel = target != null ? target.getVelocity() : Vec3d.ZERO;
            if (vel.lengthSquared() < 0.01) return last.pos;
            double smooth = 0.7;
            return last.pos.add(vel.multiply(0.05).multiply(smooth));
        }
    }

    private static class ClusterBasedModel implements PredictionModel {
        @Override
        public Vec3d predict(Deque<HistoryEntry> history, LivingEntity target) {
            if (history.size() < 3) return target != null ? target.getPos() : Vec3d.ZERO;
            List<HistoryEntry> list = new ArrayList<>(history);
            int n = list.size();
            Vec3d avgVel = Vec3d.ZERO;
            for (int i = n-3; i < n-1; i++) {
                HistoryEntry a = list.get(i);
                HistoryEntry b = list.get(i+1);
                double dt = b.dt;
                if (dt > 0) avgVel = avgVel.add(b.pos.subtract(a.pos).multiply(1.0/dt));
            }
            avgVel = avgVel.multiply(1.0/3);
            return list.get(n-1).pos.add(avgVel.multiply(0.05));
        }
    }

    private static class RecurrentModel implements PredictionModel {
        private final double[] weights = {0.5, 0.3, 0.2};
        @Override
        public Vec3d predict(Deque<HistoryEntry> history, LivingEntity target) {
            if (history.size() < 3) return target != null ? target.getPos() : Vec3d.ZERO;
            List<HistoryEntry> list = new ArrayList<>(history);
            int n = list.size();
            Vec3d predOffset = Vec3d.ZERO;
            double totalW = 0;
            for (int i = 0; i < Math.min(weights.length, n-1); i++) {
                HistoryEntry a = list.get(n-2-i);
                HistoryEntry b = list.get(n-1-i);
                double dt = b.dt;
                if (dt > 0) {
                    Vec3d offset = b.pos.subtract(a.pos);
                    predOffset = predOffset.add(offset.multiply(weights[i]));
                    totalW += weights[i];
                }
            }
            if (totalW > 0) predOffset = predOffset.multiply(1.0/totalW);
            return list.get(n-1).pos.add(predOffset);
        }
    }

    private static class PlayerProfile {
        private final String name;
        private double avgSpeed = 0;
        private double speedVariance = 0;
        private double turnFrequency = 0;
        private int samples = 0;
        private final Deque<Double> turnAngles = new ArrayDeque<>();

        public PlayerProfile(String name) {
            this.name = name;
        }

        public void update(LivingEntity target, Deque<HistoryEntry> history) {
            Vec3d vel = target.getVelocity();
            double speed = vel.length();
            if (samples == 0) {
                avgSpeed = speed;
            } else {
                double oldAvg = avgSpeed;
                avgSpeed = oldAvg + (speed - oldAvg) / (samples + 1);
                speedVariance = (speedVariance * samples + (speed - oldAvg)*(speed - avgSpeed)) / (samples + 1);
            }
            samples++;

            if (history.size() >= 2) {
                List<HistoryEntry> list = new ArrayList<>(history);
                HistoryEntry last = list.get(list.size()-1);
                HistoryEntry prev = list.get(list.size()-2);
                if (list.size() >= 3) {
                    Vec3d dirPrev = prev.pos.subtract(list.get(list.size()-3).pos).normalize();
                    Vec3d dirLast = last.pos.subtract(prev.pos).normalize();
                    double angle = Math.acos(MathHelper.clamp(dirPrev.dotProduct(dirLast), -1, 1));
                    turnAngles.add(angle);
                    while (turnAngles.size() > 10) turnAngles.removeFirst();
                    turnFrequency = turnAngles.stream().mapToDouble(a -> a).average().orElse(0);
                }
            }
        }

        public Vec3d correctPrediction(Vec3d pred) {
            if (turnFrequency > 0.5) {
                Random r = new Random();
                pred = pred.add(new Vec3d((r.nextDouble()-0.5)*0.5, (r.nextDouble()-0.5)*0.5, (r.nextDouble()-0.5)*0.5));
            }
            return pred;
        }

        public void save() {
            try {
                Path dir = Paths.get(System.getProperty("user.dir"), "run", "profiles");
                Files.createDirectories(dir);
                Path file = dir.resolve(name + ".json");
                Gson gson = new GsonBuilder().setPrettyPrinting().create();
                try (Writer writer = new FileWriter(file.toFile())) {
                    gson.toJson(this, writer);
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }

    public static class RotationFrame {
        public final long timestamp;
        public final float yaw;
        public final float pitch;
        public final boolean attack;

        public RotationFrame(long timestamp, float yaw, float pitch, boolean attack) {
            this.timestamp = timestamp;
            this.yaw = yaw;
            this.pitch = pitch;
            this.attack = attack;
        }
    }

    public class NeuroRecorder {
        private final List<RotationFrame> frames = new ArrayList<>();
        private long startTime = 0;
        private boolean recording = false;
        private String currentRecordingName = null;

        private boolean playing = false;
        private List<RotationFrame> playbackFrames = null;
        private long playbackStart = 0;
        private int playbackIndex = 0;

        private final Path neuroDir = Paths.get(System.getProperty("user.dir"), "run", "neuro");
        private final Gson gson = new GsonBuilder().setPrettyPrinting().create();

        public NeuroRecorder() {
            try {
                Files.createDirectories(neuroDir);
            } catch (IOException e) {
                e.printStackTrace();
            }
        }

        public synchronized void startRecording(String name) {
            if (recording) stopRecording();
            frames.clear();
            recording = true;
            currentRecordingName = name;
            startTime = System.currentTimeMillis();
        }

        public synchronized void stopRecording() {
            if (!recording) return;
            if (currentRecordingName != null) {
                saveToFile(currentRecordingName);
            }
            recording = false;
            currentRecordingName = null;
        }

        public synchronized void recordFrame(float yaw, float pitch, boolean attack) {
            if (!recording) return;
            long now = System.currentTimeMillis() - startTime;
            frames.add(new RotationFrame(now, yaw, pitch, attack));
        }

        public synchronized void startPlayback(String name) {
            if (playing) stopPlayback();
            playbackFrames = loadFromFile(name);
            if (playbackFrames == null || playbackFrames.isEmpty()) return;
            playing = true;
            playbackStart = System.currentTimeMillis();
            playbackIndex = 0;
        }

        public synchronized void stopPlayback() {
            playing = false;
            playbackFrames = null;
            playbackIndex = 0;
        }

        public synchronized RotationFrame getCurrentPlaybackFrame() {
            if (!playing || playbackFrames == null) return null;
            long elapsed = System.currentTimeMillis() - playbackStart;
            while (playbackIndex < playbackFrames.size() && playbackFrames.get(playbackIndex).timestamp <= elapsed) {
                playbackIndex++;
            }
            if (playbackIndex >= playbackFrames.size()) {
                stopPlayback();
                return null;
            }
            return playbackFrames.get(playbackIndex);
        }

        public synchronized boolean isRecording() { return recording; }
        public synchronized boolean isPlaying() { return playing; }
        public synchronized String getLastRecordingName() { return currentRecordingName; }

        private void saveToFile(String name) {
            if (name == null) return;
            try {
                Path file = neuroDir.resolve(name + ".json");
                Files.createDirectories(neuroDir);
                try (Writer writer = new FileWriter(file.toFile())) {
                    gson.toJson(frames, writer);
                }
                // Убрал вывод в чат, чтобы избежать спама. Сообщение выводится при остановке.
            } catch (IOException e) {
                e.printStackTrace();
            }
        }

        private List<RotationFrame> loadFromFile(String name) {
            Path file = neuroDir.resolve(name + ".json");
            if (!Files.exists(file)) return null;
            try (Reader reader = new FileReader(file.toFile())) {
                java.lang.reflect.Type listType = new TypeToken<List<RotationFrame>>(){}.getType();
                return gson.fromJson(reader, listType);
            } catch (IOException e) {
                e.printStackTrace();
                return null;
            }
        }

        public List<String> listRecordings() {
            List<String> list = new ArrayList<>();
            try {
                Files.list(neuroDir).filter(p -> p.toString().endsWith(".json")).forEach(p -> {
                    String name = p.getFileName().toString().replace(".json", "");
                    list.add(name);
                });
            } catch (IOException e) {
                e.printStackTrace();
            }
            return list;
        }

        public void deleteRecording(String name) {
            Path file = neuroDir.resolve(name + ".json");
            try {
                Files.deleteIfExists(file);
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
